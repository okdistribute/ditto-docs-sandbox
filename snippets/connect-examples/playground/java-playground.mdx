try {
    DittoDependencies androidDependencies = new DefaultAndroidDittoDependencies(this.context);
    DittoIdentity identity = new DittoIdentity.OnlinePlayground(
        androidDependencies,
        "REPLACE_ME_WITH_YOUR_APP_ID",
        "REPLACE_ME_WITH_YOUR_PLAYGROUND_TOKEN",
        false, // This is required to be set to false to use the correct URLs
        "REPLACE_ME_WITH_YOUR_AUTH_URL",
    );

    Ditto ditto = new Ditto(androidDependencies, identity);

    DittoTransportConfig config = new DittoTransportConfig();
    config.getConnect().getWebsocketUrls().add("wss://REPLACE_ME_WITH_YOUR_WEBSOCKET_URL");

    // Optionally enable all P2P transports if using P2P Sync
    // Do not call this if only using Ditto Cloud Sync
    config.enableAllPeerToPeer();

    ditto.setTransportConfig(config);
    ditto.startSync()
} catch (DittoError e) {
    // Handle error in Ditto initialization
}
